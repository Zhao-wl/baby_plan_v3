import React, { useState, useEffect } from 'react';
import { 
  Moon, 
  Sun, 
  Utensils, 
  Smile, 
  Droplet, 
  Sparkles, 
  Home, 
  List, 
  PieChart, 
  User,
  ChevronRight,
  X,

  Clock

  const [activeTab, setActiveTab] = useState('home');
  const [showRecordModal, setShowRecordModal] = useState(false);
  const [modalType, setModalType] = useState(''); // 'eat', 'play', 'sleep', 'diaper'
  const [toast, setToast] = useState('');

  const [timer, setTimer] = useState(5130); // 1小时25分30秒的秒数

    const interval = setInterval(() => setTimer(t => t + 1), 1000);
    return () => clearInterval(interval);
  }, []);

  const formatTime = (totalSeconds) => {
    const h = Math.floor(totalSeconds / 3600);
    const m = Math.floor((totalSeconds % 3600) / 60);
    const s = totalSeconds % 60;
    return `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
  };

  const handleActionClick = (type) => {
    setModalType(type);
    setShowRecordModal(true);
  };

    setShowRecordModal(false);
    setToast('记录保存成功！');
    setTimeout(() => setToast(''), 2000);
  };

  return (
    <div className="max-w-md mx-auto h-screen bg-slate-50 flex flex-col relative overflow-hidden font-sans text-slate-800 shadow-xl border-x border-slate-100">
      
      {/* 顶部个人信息区 */}
      <div className="pt-12 pb-4 px-6 bg-white/60 backdrop-blur-md z-10 sticky top-0 flex justify-between items-center">
        <div className="flex items-center gap-4">
          <div className="w-12 h-12 bg-teal-100 rounded-full flex items-center justify-center shadow-sm overflow-hidden border-2 border-white">
            <span className="text-2xl">👶</span>
          </div>
          <div>
            <h1 className="text-lg font-bold text-slate-800 leading-tight">柚子宝宝</h1>
            <p className="text-xs text-slate-500 font-medium">3个月12天 • 6.5kg • 62cm</p>
          </div>
        </div>
        <div className="w-8 h-8 bg-white rounded-full flex items-center justify-center shadow-sm text-slate-400">
          <User size={18} />
        </div>
      </div>

      {/* 核心内容区 - 可滚动 */}
      <div className="flex-1 overflow-y-auto px-6 pb-40 scrollbar-hide">
        
        {/* 1. 实时计时器卡片 */}
        <div className="mt-4 bg-gradient-to-br from-blue-50 to-indigo-50 rounded-[32px] p-6 shadow-sm border border-blue-100/50 relative overflow-hidden flex flex-col items-center justify-center min-h-[160px]">
          {/* 背景呼吸波纹 */}
          <div className="absolute w-32 h-32 bg-blue-200/30 rounded-full animate-ping" style={{ animationDuration: '3s' }}></div>
          <div className="absolute w-48 h-48 bg-blue-100/20 rounded-full animate-pulse" style={{ animationDuration: '4s' }}></div>
          
          <div className="relative z-10 flex flex-col items-center">
            <div className="bg-white/60 backdrop-blur-sm px-4 py-1.5 rounded-full mb-3 flex items-center gap-2">
              <Moon size={16} className="text-indigo-500" />
              <span className="text-sm font-semibold text-indigo-700">正在睡眠中...</span>
            </div>
            <div className="text-4xl font-black text-slate-700 tracking-wider">
              {formatTime(timer)}
            </div>
            <p className="text-xs text-slate-400 mt-2">今日累计睡眠: 4h 20m</p>
          </div>
        </div>

        {/* 2. 🤖 智能行为预测卡片 */}
        <div className="mt-6 bg-gradient-to-r from-purple-50 to-pink-50 rounded-3xl p-5 shadow-sm border border-pink-100/50">
          <div className="flex items-center gap-2 mb-3">
            <div className="bg-purple-100 p-1.5 rounded-full">
              <Sparkles size={16} className="text-purple-600" />
            </div>
            <h2 className="text-sm font-bold text-slate-700">智能预测</h2>
          </div>
          
          <div className="space-y-3">
            <div className="bg-white/70 rounded-2xl p-3 flex items-start gap-3 relative overflow-hidden">
              <div className="absolute left-0 top-0 bottom-0 w-1 bg-purple-400"></div>
              <div className="w-10 h-10 bg-purple-100 rounded-full flex flex-col items-center justify-center shrink-0">
                <span className="text-xs font-bold text-purple-700">14:30</span>
              </div>
              <div className="pt-1">
                <p className="text-sm font-semibold text-slate-700">预计即将醒来</p>
                <p className="text-xs text-slate-500 mt-0.5">醒后可能会有便便哦 💩，建议提前准备尿不湿。</p>
              </div>
            </div>
          </div>
        </div>

        {/* 3. 最近动态 */}
        <div className="mt-8">
          <div className="flex justify-between items-end mb-4 px-1">
            <h2 className="text-base font-bold text-slate-800">最近记录</h2>
            <button className="text-xs text-teal-600 font-medium flex items-center">
              查看全部 <ChevronRight size={14} />
            </button>
          </div>
          
          <div className="space-y-3">
            <div className="bg-white rounded-3xl p-4 shadow-sm flex items-center gap-4">
              <div className="w-12 h-12 bg-green-50 rounded-2xl flex items-center justify-center shrink-0">
                <Utensils size={20} className="text-green-600" />
              </div>
              <div className="flex-1">
                <div className="flex justify-between items-start">
                  <h3 className="text-sm font-bold text-slate-700">吃奶 (E)</h3>
                  <span className="text-xs text-slate-400">12:15</span>

              </div>

                <Smile size={20} className="text-yellow-600" />
              </div>
              <div className="flex-1">
                <div className="flex justify-between items-start">
                  <h3 className="text-sm font-bold text-slate-700">活动 (A)</h3>
                  <span className="text-xs text-slate-400">12:30</span>
                </div>
                <p className="text-xs text-slate-500 mt-1">趴睡练习 • 约 20 分钟</p>
              </div>
            </div>
          </div>
        </div>

        <div className="h-10"></div>
      </div>

      {/* 悬浮操作台 (Quick Actions) - 单手易操作区 */}
      <div className="absolute bottom-[88px] left-0 right-0 px-6 pointer-events-none">
        <div className="bg-white/90 backdrop-blur-md p-3 rounded-full shadow-lg border border-slate-100 flex justify-between items-center pointer-events-auto">
          <button onClick={() => handleActionClick('eat')} className="flex flex-col items-center gap-1.5 w-16 group">
            <div className="w-12 h-12 bg-green-100 rounded-full flex items-center justify-center group-active:scale-90 transition-transform shadow-inner">
              <Utensils size={22} className="text-green-600" />
            </div>
            <span className="text-[10px] font-bold text-green-700">吃奶</span>
          </button>
          
          <button onClick={() => handleActionClick('play')} className="flex flex-col items-center gap-1.5 w-16 group">
            <div className="w-12 h-12 bg-yellow-100 rounded-full flex items-center justify-center group-active:scale-90 transition-transform shadow-inner">
              <Smile size={22} className="text-yellow-600" />
            </div>
            <span className="text-[10px] font-bold text-yellow-700">玩耍</span>
          </button>

          <button onClick={() => handleActionClick('sleep')} className="flex flex-col items-center gap-1.5 w-16 group">
            <div className="w-12 h-12 bg-blue-100 rounded-full flex items-center justify-center group-active:scale-90 transition-transform shadow-inner">
              <Moon size={22} className="text-blue-600" />
            </div>
            <span className="text-[10px] font-bold text-blue-700">睡眠</span>
          </button>

          <button onClick={() => handleActionClick('diaper')} className="flex flex-col items-center gap-1.5 w-16 group">
            <div className="w-12 h-12 bg-orange-100 rounded-full flex items-center justify-center group-active:scale-90 transition-transform shadow-inner">
              <Droplet size={22} className="text-orange-600" />
            </div>
            <span className="text-[10px] font-bold text-orange-700">排泄</span>
          </button>
        </div>
      </div>

      {/* 底部导航栏 */}
      <div className="bg-white h-20 border-t border-slate-100 flex justify-around items-center px-2 pb-safe absolute bottom-0 left-0 right-0 z-20">
        <button className={`flex flex-col items-center gap-1 p-2 ${activeTab === 'home' ? 'text-teal-600' : 'text-slate-400'}`} onClick={() => setActiveTab('home')}>
          <Home size={24} strokeWidth={activeTab === 'home' ? 2.5 : 2} />
          <span className="text-[10px] font-bold">首页</span>
        </button>
        <button className={`flex flex-col items-center gap-1 p-2 ${activeTab === 'timeline' ? 'text-teal-600' : 'text-slate-400'}`} onClick={() => setActiveTab('timeline')}>
          <List size={24} strokeWidth={activeTab === 'timeline' ? 2.5 : 2} />
          <span className="text-[10px] font-bold">时间线</span>
        </button>
        <button className={`flex flex-col items-center gap-1 p-2 ${activeTab === 'stats' ? 'text-teal-600' : 'text-slate-400'}`} onClick={() => setActiveTab('stats')}>
          <PieChart size={24} strokeWidth={activeTab === 'stats' ? 2.5 : 2} />
          <span className="text-[10px] font-bold">统计</span>
        </button>
      </div>

      {/* 快捷记录弹窗 (模拟从底部弹出的半屏交互) */}
      {showRecordModal && (
        <div className="absolute inset-0 z-50 flex flex-col justify-end">
          <div className="absolute inset-0 bg-slate-900/40 backdrop-blur-sm" onClick={() => setShowRecordModal(false)}></div>

            <div className="w-12 h-1.5 bg-slate-200 rounded-full mx-auto mb-6"></div>
            
            <div className="flex justify-between items-center mb-6">
              <h2 className="text-xl font-bold text-slate-800">

              </h2>
              <button onClick={() => setShowRecordModal(false)} className="bg-slate-100 p-2 rounded-full text-slate-500">
                <X size={20} />
              </button>
            </div>

            {/* 这里以“吃奶”为例展示低认知负荷的录入方式 */}
            {modalType === 'eat' && (
              <div className="space-y-5">
                <div>
                  <label className="text-sm font-bold text-slate-600 mb-2 block">喂养方式 (单选)</label>
                  <div className="flex gap-3">
                    <button className="flex-1 py-3 px-4 bg-teal-50 border-2 border-teal-500 text-teal-700 rounded-2xl font-bold flex justify-center items-center gap-2">
                      <CheckCircle2 size={18} /> 母乳亲喂
                    </button>
                    <button className="flex-1 py-3 px-4 bg-slate-50 border-2 border-slate-100 text-slate-600 rounded-2xl font-bold">
                      瓶喂奶粉
                    </button>
                  </div>
                </div>

                <div>
                  <label className="text-sm font-bold text-slate-600 mb-2 block">开始时间</label>
                  <div className="flex items-center gap-3 bg-slate-50 p-4 rounded-2xl border border-slate-100">
                    <Clock size={20} className="text-slate-400" />
                    <span className="text-slate-700 font-bold text-lg">当前 (14:15)</span>
                    <span className="ml-auto text-sm text-teal-600 font-medium">修改</span>
                  </div>
                </div>
              </div>
            )}

            <button 
              onClick={handleSaveRecord}
              className="w-full mt-8 bg-teal-600 text-white font-bold py-4 rounded-2xl shadow-lg shadow-teal-200 flex justify-center items-center gap-2 active:scale-[0.98] transition-transform"
            >
              保存记录
            </button>
          </div>
        </div>
      )}

      {/* Toast 提示 */}
      {toast && (
        <div className="absolute top-20 left-1/2 transform -translate-x-1/2 bg-slate-800 text-white px-4 py-2 rounded-full shadow-lg text-sm font-medium z-50 flex items-center gap-2 animate-fade-in">
          <CheckCircle2 size={16} className="text-green-400" /> {toast}
        </div>
      )}

      <style dangerouslySetInnerHTML={{__html: `
        @keyframes slide-up {
          from { transform: translateY(100%); }
          to { transform: translateY(0); }
        }
        .animate-slide-up {
          animation: slide-up 0.3s cubic-bezier(0.16, 1, 0.3, 1);
        }
        @keyframes fade-in {
          from { opacity: 0; transform: translate(-50%, -10px); }
          to { opacity: 1; transform: translate(-50%, 0); }
        }
        .animate-fade-in {
          animation: fade-in 0.2s ease-out;
        }
        .scrollbar-hide::-webkit-scrollbar {
          display: none;
        }
        .scrollbar-hide {
          -ms-overflow-style: none;
          scrollbar-width: none;
        }
      `}} />
    </div>
  );
}

**原型设计亮点说明：**
1. 首页卡片化与呼吸动效：最醒目的位置留给了当前正在进行的状态（示例中为睡眠），背景设计了轻微的呼吸波纹（Pulse），传达出“生命正在悄悄生长”的治愈感。

2. 智能预测模块直观展示：在“智能预测”卡片中，实现了 PRD 中“合并相近行为”的理念（如展示“预计即将醒来，醒后可能会有便便”），视觉上使用了微渐变和特殊的图标，将其与其他历史记录区分开。

3. 单手快捷操作台：有别于传统的加号按钮，我在屏幕底部（手指最容易触达的舒适区）设计了一个弧形的悬浮操作台，分别对应 E(吃)、A(玩)、S(睡)、排泄。您可以尝试点击“吃奶”按钮。

4. 低认知负荷的弹窗体验：点击“吃奶”后弹出的记录框，遵循了 PRD 中“尽量使用选型而非输入”的原则，提供了巨大、清晰的选项块（Chips），在抱娃慌乱时也能精准点击。
